-- ============================================================ -- mo.co tracker — get_player(): rich per-hunter dashboard data. -- Run ONCE in the mo.co project's SQL editor. Safe to re-run. -- Powers the Profile tab + leaderboard player lookup. -- Returns, for one handle (hidden rows excluded): -- totals, rank, by_type, tier breakdown, core breakdown, -- streaks (current/longest dry run, best multi-drop), -- a community baseline (rate + per-core) for the luck meter, -- and recent kills (with ids, for self-delete). -- ============================================================ create or replace function get_player(h text) returns json language sql stable security definer set search_path = public as $$ with mine as ( select id, difficulty, drops, jsonb_array_length(drops) nd, created_at from logs where lower(coalesce(handle,'')) = lower(h) and not hidden ), seq as ( select nd, row_number() over (order by created_at, id) rn from mine ), islands as ( -- gaps-and-islands: constant g within each run of equal zero-status select rn, nd, rn - row_number() over (partition by (nd = 0) order by rn) g from seq ), dr as ( select e->>'tier' tier, e->>'core' core from mine, lateral jsonb_array_elements(drops) e ), ranks as ( select handle, row_number() over (order by count(*) desc) rn from logs where handle is not null and handle <> '' and not hidden group by handle ), capped as ( -- community, per-handle capped to match get_stats select drops from ( select drops, row_number() over (partition by coalesce(handle,'') order by created_at, id) rn2 from logs where not hidden ) q where rn2 <= 3000 ), comm_cores as ( select e->>'core' core, count(*)::int c from capped, lateral jsonb_array_elements(drops) e group by 1 ), paced as ( -- gaps between consecutive APP-LOGGED kills only; imports/seeds have no token, so excluded select extract(epoch from (created_at - lag(created_at) over (order by created_at, id))) g from logs where lower(coalesce(handle,'')) = lower(h) and not hidden and token is not null ) select json_build_object( 'handle', h, 'kills', (select count(*)::int from mine), 'drops', (select coalesce(sum(nd),0)::int from mine), 'with_drops', (select count(*)::int from mine where nd > 0), 'pace_per_min', (select case when sg > 0 and kg >= 3 then round((kg/(sg/60.0))::numeric,1) else null end from (select count(*) filter (where g>=2 and g<=300) kg, sum(g) filter (where g>=2 and g<=300) sg from paced) p), 'rank', (select rn::int from ranks where lower(handle) = lower(h) limit 1), 'contributors', (select count(*)::int from ranks), 'by_type', (select coalesce(jsonb_object_agg(difficulty,c),'{}'::jsonb) from (select difficulty, count(*)::int c from mine group by difficulty) a), 'tiers', (select coalesce(jsonb_object_agg(tier,c),'{}'::jsonb) from (select tier, count(*)::int c from dr group by tier) b), 'cores', (select coalesce(jsonb_object_agg(core,c),'{}'::jsonb) from (select core, count(*)::int c from dr group by core) d), 'streaks', json_build_object( 'best_multi', (select coalesce(max(nd),0)::int from mine), 'current_dry', (select count(*)::int from seq where rn > coalesce((select max(rn) from seq where nd > 0), 0)), 'longest_dry', (select coalesce(max(cnt),0)::int from (select count(*) cnt from islands where nd = 0 group by g) z) ), 'community', json_build_object( 'kills', (select count(*)::int from capped), 'with_drops', (select count(*)::int from capped where jsonb_array_length(drops) > 0), 'cores', (select coalesce(jsonb_object_agg(core,c),'{}'::jsonb) from comm_cores) ), 'recent', (select coalesce(json_agg(json_build_object('id',id,'difficulty',difficulty,'drops',drops)),'[]'::json) from (select id, difficulty, drops, created_at from mine order by created_at desc limit 30) r) ); $$; grant execute on function get_player(text) to anon, authenticated;
